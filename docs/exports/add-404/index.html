    404页面

    配置 404 页面

    @tuya/dumi-theme-tuya 提供了内置 404 页面组件，可以直接使用，在 src/app.ts 中添加代码：

    // src/app.ts
    import { NotFound } from '@tuya/dumi-theme-tuya/exports';
    

    export function patchRoutes({ routes }) {
      routes[2].routes.push({
        path: '*',
        component: NotFound,
      });
    }
